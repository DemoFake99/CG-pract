#include <graphics.h>
#include <conio.h>
int main(void)
{
   int gdriver = DETECT, gmode, errorcode;
   initgraph(&gdriver, &gmode, "c:\\turboc3\\bgi");

   rectangle(150,180,250,300);
   rectangle(250,180,420,300);

   line(200,100,370,100);
   line(370,100,420,180);

   getch();

}
