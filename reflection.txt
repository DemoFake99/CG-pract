#include<graphics.h>
#include<stdio.h>
#include<conio.h>

{
	int gd = DETECT, gm;
	int x1=100, y1=100, x2=200, y2=100, x3=200, y3=200;
	int dfx, dfy;
	initgraph(&gd, &gm, "c:\\turboc3\\bgi");
	
	line(getmaxx() / 2, 0, getmaxx() / 2,getmaxy());
